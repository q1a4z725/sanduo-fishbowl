function updateAndDrawFish(fish, idx) { let pos = fishPositions[idx]; if (!pos) return; pos.offX += pos.vx; pos.offY += pos.vy; // 以第一筆劃的第一個點作為基準來判斷邊界 let firstPoint = fish.strokes[0].points[0]; let headX = firstPoint.x + pos.offX; let headY = firstPoint.y + pos.offY; if (headX < 0 || headX > width) pos.vx *= -1; if (headY < 0 || headY > height) pos.vy *= -1; push(); translate(pos.offX, pos.offY); if (pos.vx < 0) { translate(firstPoint.x * 2, 0); scale(-1, 1); } // 繪製這條魚的所有筆劃 for (let s of fish.strokes) { stroke(s.color); strokeWeight(3); noFill(); beginShape(); for (let j = 0; j < s.points.length; j++) { let swing = sin(frameCount * 0.1 + j * 0.3) * (j * 0.2); vertex(s.points[j].x + swing, s.points[j].y + swing * 0.3); } endShape(); } pop(); }
